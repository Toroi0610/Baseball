def make_prob_matrix(table):
    """
    ---------
    input
    ---------
    table9Pandas.DataFrame
    
    ---------
    output
    ---------
    P 9 *  25 *  25
    """

    PA = table["PA"].values - table["EOB"].values
    dou = table["DOU"].values
    tri = table["TRI"].values
    hom = table["HR"].values
    wal = table["BB"].values + table["HBP"].values
    sin = table["H"].values - dou - tri - hom
    sh = table["SH"].values
    sf = table["SF"].values


    pw = wal / PA # (4)
    ps = sin / PA # (5)
    pd = dou / PA # (6)
    pt = tri / PA # (7)
    ph = hom / PA # (8)
    psh = sh / PA # (9)
    psf = sf / PA # (10)

    po = 1 - pw - ps - pd - pt - ph - psh - psf # (11)


    P = np.zeros([9, 25, 25])

    for i in range(9):

        # (13)
        A = np.array([[ph[i], ps[i]+pw[i], pd[i], pt[i], 0, 0, 0, 0],
                      [ph[i], 0, 0, pt[i], ps[i]+pw[i], 0, pd[i], 0],
                      [ph[i], ps[i], pd[i], pt[i], pw[i], 0, 0, 0],
                      [ph[i], ps[i], pd[i], pt[i], 0, pw[i], 0, 0],
                      [ph[i], 0, 0, pt[i], ps[i], 0, pd[i], pw[i]],
                      [ph[i], 0, 0, pt[i], ps[i], 0, pd[i], pw[i]],
                      [ph[i], ps[i], pd[i], pt[i], 0, 0, 0, pw[i]],
                      [ph[i], 0, 0, pt[i], ps[i], 0, pd[i], pw[i]]])

        # (14)
        B = np.identity(8) * (po[i] + psh[i])
        # (15)
        F = np.ones([8]) * (po[i] + psh[i] + psf[i])

        P[i, :8, :8] = A
        P[i, :8, 8:16] = B
        P[i, 8:16, 8:16] = A
        P[i, 8:16, 16:24] = B
        P[i, 16:24, 16:24] = A
        P[i, 16:24, 24] = F
        P[i, 24, 24] = 1

        P[i, 1, 10] = psh[i]
        P[i, 2, 11] = psf[i]
        P[i, 3, 8] = psf[i]
        P[i, 4, 13] = psf[i]
        P[i, 5, 9] = psf[i]
        P[i, 6, 11] = psf[i]
        P[i, 7, 13] = psf[i]
        P[i, 9, 18] = psh[i]
        P[i, 10, 19] = psf[i]
        P[i, 11, 16] = psf[i]
        P[i, 12, 21] = psf[i]
        P[i, 13, 17] = psf[i]
        P[i, 14, 19] = psf[i]
        P[i, 15, 21] = psf[i]

        P[i, 0, 8] += psf[i]
        P[i, 8, 16] += psf[i]
        P[i, 1, 9] += psf[i] - psh[i]
        P[i, 9, 17] += psf[i] - psh[i]

    return P
